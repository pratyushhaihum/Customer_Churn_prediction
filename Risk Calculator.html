Database Schema & Sample Data

Required Tables Schema

-- Customer Data Table
CREATE TABLE customers (
    customer_id INT PRIMARY KEY,
    age INT,
    gender VARCHAR(10),
    tenure INT,
    monthly_charges DECIMAL(10,2),
    total_charges DECIMAL(10,2),
    contract_type VARCHAR(20),
    payment_method VARCHAR(30),
    internet_service VARCHAR(20),
    phone_service BOOLEAN,
    multiple_lines BOOLEAN,
    online_security BOOLEAN,
    online_backup BOOLEAN,
    device_protection BOOLEAN,
    tech_support BOOLEAN,
    streaming_tv BOOLEAN,
    streaming_movies BOOLEAN,
    paperless_billing BOOLEAN,
    churn_status BOOLEAN,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Usage Patterns Table
CREATE TABLE usage_patterns (
    usage_id INT PRIMARY KEY,
    customer_id INT,
    data_usage_gb DECIMAL(10,2),
    call_minutes INT,
    sms_count INT,
    support_tickets INT,
    login_frequency INT,
    last_activity_date DATE,
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id)
);

-- Predictions Table
CREATE TABLE churn_predictions (
    prediction_id INT PRIMARY KEY,
    customer_id INT,
    churn_probability DECIMAL(5,4),
    risk_level VARCHAR(10),
    prediction_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    model_version VARCHAR(20),
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id)
);
